cmake_minimum_required(VERSION 3.12)
project(TestRadixSort LANGUAGES CXX CUDA)

find_package(CUDA REQUIRED)

# Create test executable
add_executable(test_radix_sort 
    test_radix_sort.cu
    radixsort.cu
    radixsort.cpp
)

# Set CUDA properties
set_property(TARGET test_radix_sort PROPERTY CUDA_SEPARABLE_COMPILATION ON)
set_target_properties(test_radix_sort PROPERTIES CUDA_ARCHITECTURES "50;60;70;75;80")

# Link CUDA libraries
target_link_libraries(test_radix_sort ${CUDA_LIBRARIES})

# Include directories
target_include_directories(test_radix_sort PRIVATE ${CUDA_INCLUDE_DIRS})
